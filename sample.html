サンプルソース

基本の処理

<script type="text/javascript">
	$(window).load(function(){
		$.forwardpop
			.init({
				defMinWidth: 500,//デフォルトの最小幅
				tpl:function(src, minWidth, title, item){//ポップアップウィンドウのテンプレートメソッド
					var fin = '';
					fin += '<!doctype html>';
					fin += '<html>';
					fin += '<head>';
					fin += '<meta charset="UTF-8" />';
					fin += '<meta name="viewport" content="width='+(20+minWidth)+'" />';
					fin += '<title>'+$('<div/>').text( title ).html()+' | jquery.forwardpop.js | DEMO</title>';
					fin += '<link rel="stylesheet" href="styles/setup.css" />'+"\n";
					fin += '<link rel="stylesheet" href="styles/modules.css" />'+"\n";
					fin += '</head>';
					fin += '<body>';
					fin += '<div id="content" class="contents">';
					fin += '<div style="width:'+minWidth+'px; margin:0 auto;">'+src+'</div>';
					fin += '</div>';
					fin += '</body>';
					fin += '</html>';
					return fin;
				},
				defTitle: 'コンテンツ'//コンテンツのデフォルトのタイトル
			})
			.addElements('.forwardpop')
		;
	});
</script>
<div class="forwardpop">
	<!-- フィットさせるコンテンツ -->
</div>

.addElements() の第2引数による最小幅の変更

<script type="text/javascript">
	$(window).load(function(){
		$.forwardpop
			.init(500,1)
			.addElements('.forwardpop')
			.addElements('.forwardpop400', 400)//第2引数で最小幅を上書きできる
		;
	});
</script>
<div class="forwardpop400">
	<!-- フィットさせるコンテンツ -->
</div>

属性 data-min-width による最小幅の変更

<div class="forwardpop" data-min-width="700">
	<!-- フィットさせるコンテンツ -->
	<!-- data-min-width による指定が最も強い -->
</div>
